title Smart Elevator Control

buildings [icon: building, color: green] {
  building_id serial pk
  name varchar 
  location text
  type enum(mall, airport, office, etc) 
}

floors [icon: life-buoy, color: blue] {
  floor_id serial pk
  building_id int fk
  floor_number int not null
}

elevator_shafts [icon: elevenlabs, color: blue] {
  shaft_id serial pk
  building_id int fk
  shaft_number int 
}

elevators [icon: electron, color: purple] {
  elevator_id serial pk
  shaft_id int fk
  building_id int fk
  capacity int not null
  current_floor_id int fk
  installation_date date 
}

elevator_floor [icon: activity-square, color: red] {
  elevator_floor_id serial pk
  elevator_id int fk
  floor_id int fk
}

floor_requests [icon: react-query, color: green] {
  request_id serial pk
  building_id int fk 
  floor_id int fk
  direction enum(up, down)
  request_time timestamp 
  status enum(pending, assigned, completed) 
}

ride_assignments [icon: aws-signer, color:red] {
  assignment_id serial pk
  request_id int fk
  elevator_id int fk
  assigned_time timestamp
}

ride_logs [icon: log, color:green] {
  ride_id serial pk
  elevator_id int fk
  request_id int fk
  start_floor_id int fk
  end_floor_id int fk
  start_time timestamp
  end_time timestamp
}

elevator_status_log [icon: edge, color: purple] {
  status_id serial pk
  elevator_id int fk
  status enum(idle, moving, maintenance, offline)
  recorded_at timestamp
}

maintenance_system [icon: safari, color: blue] {
  maintenance_id serial pk
  elevator_id int fk
  issue_desc text
  start_time timestamp
  end_time timestamp
  status enum(open, in_progress, resolved)
}


// relantionships
buildings.building_id < floors.building_id
buildings.building_id < elevators.building_id
elevators.elevator_id <> elevator_floor.elevator_id
floors.floor_id < floor_requests.floor_id
floor_requests.request_id - ride_assignments.request_id
elevators.elevator_id < ride_logs.elevator_id
elevators.elevator_id < elevator_status_log.elevator_id
elevators.elevator_id < maintenance_system.elevator_id
buildings.building_id < elevator_shafts.building_id
